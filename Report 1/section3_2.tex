\subsection{Đề xuất giải pháp sơ bộ (Proposed Solution):}
\subsubsection{Ý tưởng giải pháp}
Từ các pain points đã xác định, nhóm đề xuất một giải pháp nhằm cải thiện trải nghiệm học tập và giải quyết các vấn đề cốt lõi của người học.
\textbf{Tên giải pháp: AI RPG English Learning System}
\begin{itemize}[label=-]
\item \textbf{Mô tả ngắn:} 
\begin{itemize}[label=+]
\item Giải pháp được xây dựng dưới dạng một hệ thống học tiếng Anh tích hợp trí tuệ nhân tạo, kết hợp với mô hình \textbf{game nhập vai (RPG)}. Trong đó, người học sẽ tương tác với các nhân vật (NPC) để thực hiện nhiệm vụ, từ đó học từ vựng và ngữ pháp trong bối cảnh cụ thể.
\item Khác với các phương pháp học truyền thống, hệ thống không chỉ cung cấp kiến thức mà còn tạo điều kiện để người học \textbf{vận dụng kiến thức ngay trong quá trình học}, thông qua các tình huống mô phỏng thực tế.
\end{itemize}
\item \textbf{Tính mới:} Giải pháp kết hợp ba thành phần chính:
\begin{itemize}[label=+]
\item \textbf{AI hội thoại (LLM):}Tạo phản hồi tự nhiên và sửa lỗi theo thời gian thực.
\item \textbf{Game hóa (RPG):}Tạo động lực và duy trì sự hứng thú.
\item \textbf{Learning Loop:}Cá nhân hóa lộ trình học.
\end{itemize}

\item \textbf{Cách giải quyết Pain Points (Mapping):}
\begin{itemize}[label=+]
\item \textbf{Pain Point 1: Thiếu phản hồi theo thời gian thực.}

\textbf{$\rightarrow$ AI Feedback Engine:} phát hiện và sửa lỗi ngay trong hội thoại.
\item \textbf{Pain Point 2: Áp lực khi diễn đạt.}

$\rightarrow$ \textbf{Role-play với NPC:} tạo môi trường luyện tập an toàn, không áp lực.
\item \textbf{Pain Point 3: Thiếu ngữ cảnh thực hành.}

$\rightarrow$ \textbf{Quest-based Learning:} học thông qua nhiệm vụ và tình huống cụ thể.
\end{itemize}
\end{itemize}

\subsubsection{Các tính năng chính (Core Features)}
\begin{itemize}[label=-]
\item \textbf{Hệ thống Phản hồi AI Tức thời (Real-time AI Feedback Engine):} 

\hspace{0.5cm} \textbf{Mô tả:} Khi người chơi tương tác với NPC thông qua việc chọn đáp án từ vựng/ngữ pháp hoặc điền từ vào chỗ trống (dựa trên bài đọc), hệ thống AI LLM sẽ lập tức đánh giá đúng/sai và đưa ra lời giải thích chi tiết về quy tắc ngữ pháp hoặc ngữ nghĩa của từ vựng đó ngay trên màn hình.

\hspace{0.5cm} \textbf{Giải quyết:} Pain Point 1 (Thiếu phản hồi $\&$ sửa lỗi theo thời gian thực). Nó giúp giải quyết tình trạng người học không nhận diện và sửa lỗi kịp thời. 
\item \textbf{Hội thoại Phân nhánh theo Ngữ cảnh (Contextual Choice-Based Dialogue)} 

\hspace{0.5cm} \textbf{Mô tả:} Người chơi không cần gõ câu trả lời (loại bỏ Writing). Thay vào đó, họ phải đọc hiểu (Reading) lời thoại của NPC và đưa ra quyết định bằng cách chọn các câu phản hồi có sẵn. Các lựa chọn này chứa từ vựng hoặc cấu trúc ngữ pháp mục tiêu. Chọn đúng ngữ cảnh sẽ giúp hội thoại tiếp tục trơn tru.

\hspace{0.5cm} \textbf{Giải quyết:} Pain Point 2 (Áp lực khi tự viết câu, thiếu tự tin). Bằng cách này, người chơi vẫn học được cách sử dụng từ trong câu mà không bị rào cản tâm lý khi phải tự "sản xuất" (produce) ngôn ngữ.
\item \textbf{Học tập dựa trên Nhiệm vụ (Quest-based Learning System)}

\hspace{0.5cm} \textbf{Mô tả:} Từ vựng và bài tập đọc hiểu không xuất hiện dưới dạng danh sách khô khan, mà được lồng ghép vào các nhiệm vụ RPG (ví dụ: đọc một bức thư cổ để tìm mật mã, hoặc chọn đúng từ vựng phép thuật để mở cửa). Hoàn thành nhiệm vụ sẽ được thưởng EXP và vật phẩm.

\hspace{0.5cm} \textbf{Giải quyết:} Pain Point 3 (Học thụ động, thiếu môi trường thực hành). Giúp người học áp dụng kiến thức vào các tình huống mô phỏng thực tế , xóa bỏ sự nhàm chán của việc lặp đi lặp lại một kiểu câu hỏi.
\end{itemize}

\textbf{Công thức tính mức độ ưu tiên rủi ro:}
Dựa vào điểm số (Score), mức độ ưu tiên xử lý rủi ro được phân loại như sau:
\begin{itemize}[label=$\bullet$]
    \item \textbf{Score $\geq$ 12:} Ưu tiên cao (Cần có biện pháp xử lý và theo dõi sát sao).
    \item \textbf{Score từ 8 - 11:} Ưu tiên trung bình (Cần chuẩn bị sẵn phương án dự phòng).
    \item \textbf{Score $<$ 8:} Ưu tiên thấp (Chấp nhận rủi ro và theo dõi định kỳ).
\end{itemize}
\subsection{Tiêu chí thành công (Success Criteria)}
\subsubsection{Chỉ số định lượng (Quantitative Metrics)}
\begin{itemize}[label=-]
\item \textbf{Hiệu suất:} Thời gian phản hồi AI (sửa lỗi + sinh bài tập) $\leq$ 5 giây.
\item \textbf{Sử dụng:} $\leq$ 70$\%$ người dùng quay lại ít nhất 2 lần/tuần.
\item \textbf{Hiệu quả học tập:} Điểm bài test ngữ pháp/từ vựng tăng $\geq$ 25$\%$ sau 2 tuần sử dụng.
Hoàn thành nhiệm vụ: $\geq$ 80$\%$ quest được hoàn thành mỗi phiên chơi.
\item \textbf{Độ chính xác AI:} $\geq$ 90$\%$ câu sửa ngữ pháp được đánh giá đúng.
\item \textbf{Lỗi hệ thống:} Tỷ lệ lỗi (API fail, crash) $<$ 3$\%$.
\end{itemize}
\subsubsection{Chỉ số định tính (Qualitative Metrics)}
\begin{itemize}[label=-]
\item Người dùng cảm thấy việc học tiếng Anh \textbf{thú vị hơn so với cách truyền thống}.
\item Người học \textbf{hiểu rõ lỗi sai của mình} thay vì chỉ biết đúng/sai.
\item Trải nghiệm game giúp \textbf{tăng động lực học lâu dài}.
\item Người dùng đánh giá hệ thống \textbf{dễ sử dụng, thân thiện}.
\item Nội dung học được cảm nhận là \textbf{cá nhân hóa (đúng điểm yếu của mình)}.
\end{itemize}
\subsubsection{Cách đo lường}
\begin{itemize}[label=-]
\item \textbf{Khảo sát trước, sau khi sử dụng:} đo mức cải thiện kiến thức và mức độ hài lòng.
\item \textbf{System Logs (Firebase):} theo dõi số lần đăng nhập, thời gian sử dụng, tần suất làm quest.
\item \textbf{AI evaluation:} so sánh output AI với tập câu chuẩn để đo độ chính xác.
\item \textbf{Tracking tiến độ học:} lưu lịch sử đúng/sai để đo improvement ($\%$).
\item \textbf{Phỏng vấn người dùng:} thu thập feedback về trải nghiệm và mức độ hứng thú.
\end{itemize}
